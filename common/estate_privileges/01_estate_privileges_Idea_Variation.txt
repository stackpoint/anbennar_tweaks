	can_select = {

	}

			if = {
				limit = { 
					army_tradition = 100

			}

				limit = { 
					army_tradition = 90

				}
				create_conquistador = { tradition = 90 }
			}
			if = {
				limit = { 
					army_tradition = 80
					NOT = { army_tradition = 90 }
				}
				create_conquistador = { tradition = 80 }
			}
			if = {
				limit = { 
					army_tradition = 70
					NOT = { army_tradition = 80 }
				}
				create_conquistador = { tradition = 70 }
			}
			if = {
				limit = { 
					army_tradition = 60
					NOT = { army_tradition = 70 }
				}
				create_conquistador = { tradition = 60 }
			}
			if = {
				limit = { 
					army_tradition = 50
					NOT = { army_tradition = 60 }
				}
				create_conquistador = { tradition = 50 }
			}
			if = {
				limit = { 
					army_tradition = 0
					NOT = { army_tradition = 50 }
				}
				create_conquistador = { tradition = 40 }
			}
	}
	
	penalties = {
		nobles_loyalty_modifier = -0.01

		janissary_loyalty_modifier = -0.01

		dhimmi_loyalty_modifier = -0.01
		brahmins_hindu_loyalty_modifier = -0.01
		brahmins_muslim_loyalty_modifier = -0.01 
		brahmins_other_loyalty_modifier = -0.01
		maratha_loyalty_modifier = -0.01
		vaisyas_loyalty_modifier = -0.01
		jains_loyalty_modifier = -0.01
		rajput_loyalty_modifier = -0.01
	}
	
	cooldown_years = 10
	on_cooldown_expires = {
		remove_estate_privilege = estate_priv_idea_variation_1_11
	}
	
	ai_will_do = {
		factor = 10
